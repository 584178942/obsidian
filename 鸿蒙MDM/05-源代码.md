```typescript
// ========== MqttClient.ets ==========
import mqtt from '@ohos/mqtt'
import { MqttConfig } from './MqttConfig'

export class MqttClientManager {
  private client: mqtt.MqttClient | null = null
  private deviceId: string = ''
  private isConnected: boolean = false
  private listeners: ((connected: boolean) => void)[] = []
  private messageHandler: ((topic: string, payload: string) => void) | null = null

  initialize(deviceId: string) {
    this.deviceId = deviceId
    this.client = mqtt.createMqttClient(MqttConfig.SERVER_URI, {
      clientId: MqttConfig.CLIENT_ID_PREFIX + deviceId,
      cleanSession: MqttConfig.CLEAN_SESSION,
      keepAlive: MqttConfig.KEEP_ALIVE_INTERVAL
    })
    this.client.on('connect', this.onConnect.bind(this))
    this.client.on('message', this.onMessage.bind(this))
    this.client.on('close', this.onClose.bind(this))
    this.client.on('error', this.onError.bind(this))
  }

  setMessageHandler(handler: (topic: string, payload: string) => void) {
    this.messageHandler = handler
  }

  connect() {
    if (!this.client) return
    this.client.connect({
      username: MqttConfig.USERNAME,
      password: MqttConfig.PASSWORD,
      connectTimeout: MqttConfig.CONNECTION_TIMEOUT * 1000,
      reconnectPeriod: 5000,
      will: {
        topic: MqttConfig.WILL_TOPIC,
        payload: JSON.stringify({ action: 'offline', code: this.deviceId }),
        qos: MqttConfig.WILL_QOS,
        retained: MqttConfig.WILL_RETAINED
      }
    })
  }

  subscribe(topic: string, qos: number = MqttConfig.SUBSCRIBE_QOS) {
    this.client?.subscribe(topic, qos)
  }

  publish(topic: string, message: string, qos: number = MqttConfig.PUBLISH_QOS): boolean {
    if (!this.isConnected) return false
    this.client?.publish(topic, message, qos)
    return true
  }

  disconnect() { this.client?.disconnect() }

  isConnectedState(): boolean { return this.isConnected }
  getDeviceId(): string { return this.deviceId }

  addConnectionListener(listener: (connected: boolean) => void) {
    this.listeners.push(listener)
  }

  removeConnectionListener(listener: (connected: boolean) => void) {
    const idx = this.listeners.indexOf(listener)
    if (idx >= 0) this.listeners.splice(idx, 1)
  }

  private onConnect() {
    this.isConnected = true
    this.listeners.forEach(fn => fn(true))
    this.subscribe(`mdm/${this.deviceId}`)
    this.subscribe(`device/config/${this.deviceId}`)
    console.info('MQTT connected')
  }

  private onMessage(topic: string, payload: string) {
    console.info(`MQTT message on ${topic}`)
    if (this.messageHandler) this.messageHandler(topic, payload)
  }

  private onClose() {
    this.isConnected = false
    this.listeners.forEach(fn => fn(false))
    console.info('MQTT disconnected')
  }

  private onError(err: any) { console.error('MQTT error:', err) }
}
```

```typescript
// ========== ReconnectManager.ets ==========
export class ReconnectManager {
  private disconnectTime = 0
  private isDisconnectRecorded = false

  recordDisconnect() {
    this.disconnectTime = Date.now()
    this.isDisconnectRecorded = true
  }

  reset() {
    this.disconnectTime = 0
    this.isDisconnectRecorded = false
  }

  getDelayMs(): number {
    if (!this.isDisconnectRecorded) return 5000
    const minutes = (Date.now() - this.disconnectTime) / 60000
    if (minutes < 5) return 5000
    if (minutes < 30) return 30000
    if (minutes < 60) return 60000
    if (minutes < 360) return 180000
    return 300000
  }

  getDisconnectMinutes(): number {
    if (!this.isDisconnectRecorded) return 0
    return (Date.now() - this.disconnectTime) / 60000
  }
}
```

```typescript
// ========== CommandMessage.ets ==========
export class CommandMessage {
  commandId: string
  action: string
  params: Record<string, any> | null
  timestamp: number

  constructor(commandId: string, action: string, params: Record<string, any> | null, timestamp: number) {
    this.commandId = commandId
    this.action = action
    this.params = params
    this.timestamp = timestamp
  }

  static fromJson(json: string): CommandMessage | null {
    try {
      const obj = JSON.parse(json)
      if (!obj.commandId || !obj.action) return null
      return new CommandMessage(obj.commandId, obj.action, obj.params || null, obj.timestamp || Date.now())
    } catch { return null }
  }

  getString(key: string, def: string = ''): string {
    if (!this.params) return def
    const val = this.params[key]
    return typeof val === 'string' ? val : def
  }

  getBoolean(key: string, def: boolean = false): boolean {
    if (!this.params) return def
    const val = this.params[key]
    return typeof val === 'boolean' ? val : def
  }

  getNumber(key: string, def: number = 0): number {
    if (!this.params) return def
    const val = this.params[key]
    return typeof val === 'number' ? val : def
  }

  hasParam(key: string): boolean {
    return this.params !== null && key in this.params
  }
}
```

```typescript
export class CommandResult {
  commandId: string
  action: string
  success: boolean
  message: string
  data: Record<string, any> | null
  timestamp: number

  constructor(commandId: string, action: string, success: boolean, message: string,
               data: Record<string, any> | null = null, timestamp: number = Date.now()) {
    this.commandId = commandId
    this.action = action
    this.success = success
    this.message = message
    this.data = data
    this.timestamp = timestamp
  }

  static success(cmdId: string, action: string, msg: string = '执行成功', data?: Record<string, any>): CommandResult {
    return new CommandResult(cmdId, action, true, msg, data || null)
  }

  static failure(cmdId: string, action: string, msg: string): CommandResult {
    return new CommandResult(cmdId, action, false, msg, null)
  }

  toJson(): string { return JSON.stringify(this) }
}
```

```typescript
// ========== CommandHandler.ets ==========
import { CommandMessage } from './CommandMessage'
import { CommandResult } from './CommandResult'
import { DeviceController } from '../mdm/DeviceController'
import { AppManager } from '../mdm/AppManager'
import { RestrictionManager } from '../mdm/RestrictionManager'
import { InfoQuerier } from '../mdm/InfoQuerier'

export class CommandHandler {
  private deviceCtrl = new DeviceController()
  private appMgr = new AppManager()
  private restrictMgr = new RestrictionManager()
  private infoQuerier = new InfoQuerier()

  handle(cmd: CommandMessage): CommandResult {
    switch (cmd.action) {
      case 'lock': return this.deviceCtrl.lock(cmd)
      case 'unlock': return this.deviceCtrl.unlock(cmd)
      case 'wipe': return this.deviceCtrl.wipe(cmd)
      case 'reboot': return this.deviceCtrl.reboot(cmd)
      case 'reset_password': return this.deviceCtrl.resetPassword(cmd)
      case 'install_app': return this.appMgr.install(cmd)
      case 'uninstall_app': return this.appMgr.uninstall(cmd)
      case 'disable_wifi': return this.restrictMgr.setWifiDisabled(cmd, true)
      case 'enable_wifi': return this.restrictMgr.setWifiDisabled(cmd, false)
      case 'disable_mobile_data': return this.restrictMgr.setMobileDataDisabled(cmd, true)
      case 'enable_mobile_data': return this.restrictMgr.setMobileDataDisabled(cmd, false)
      case 'set_gps': return this.restrictMgr.setGpsEnabled(cmd, cmd.getBoolean('enabled'))
      case 'set_bluetooth': return this.restrictMgr.setBluetoothEnabled(cmd, cmd.getBoolean('enabled'))
      case 'disable_camera': return this.restrictMgr.setCameraDisabled(cmd, true)
      case 'enable_camera': return this.restrictMgr.setCameraDisabled(cmd, false)
      case 'get_device_info': return this.infoQuerier.getDeviceInfo(cmd)
      case 'get_app_list': return this.infoQuerier.getAppList(cmd)
      case 'sync_config': return this.infoQuerier.syncConfig(cmd)
      default: return CommandResult.failure(cmd.commandId, cmd.action, `未知命令: ${cmd.action}`)
    }
  }
}
```

```typescript
import battery from '@ohos.battery'
import power from '@ohos.power'
import connection from '@ohos.net.connection'

export interface HeartbeatData {
  deviceId: string
  batteryLevel: number
  isCharging: boolean
  isScreenOn: boolean
  networkType: string
  timestamp: number
}

export class HeartbeatAdjuster {
  getIntervalMs(): number {
    const info = battery.getBatteryInfo()
    const level = info.batterySOC
    const isCharging = info.chargingStatus === 1
    const isScreenOn = power.isScreenOn()
    if (isCharging && level > 50) return 60000
    if (level < 20 && !isCharging) return 300000
    if (!isScreenOn && !isCharging) return 600000
    return 120000
  }

  getHeartbeatMode(): string {
    const info = battery.getBatteryInfo()
    const level = info.batterySOC
    const isCharging = info.chargingStatus === 1
    const isScreenOn = power.isScreenOn()
    if (isCharging && level > 50) return '高性能'
    if (level < 20 && !isCharging) return '省电'
    if (!isScreenOn && !isCharging) return '超省电'
    return '普通'
  }

  buildHeartbeatMsg(deviceId: string): string {
    const info = battery.getBatteryInfo()
    const msg: HeartbeatData = {
      deviceId,
      batteryLevel: info.batterySOC,
      isCharging: info.chargingStatus === 1,
      isScreenOn: power.isScreenOn(),
      networkType: this.getNetworkType(),
      timestamp: Date.now()
    }
    return JSON.stringify(msg)
  }

  private getNetworkType(): string {
    try {
      const netHandle = connection.getDefaultNetSync()
      if (!netHandle) return 'None'
      const netCap = connection.getNetCapabilitiesSync(netHandle)
      if (netCap.bearerTypes.includes(connection.NetBearType.BEARER_WIFI)) return 'WiFi'
      if (netCap.bearerTypes.includes(connection.NetBearType.BEARER_CELLULAR)) return 'Mobile'
      return 'Unknown'
    } catch { return 'None' }
  }
}
```

```typescript
import connection from '@ohos.net.connection'

export class NetworkMonitor {
  private callback: (() => void) | null = null
  private netAvailableId: number = -1
  private netUnavailableId: number = -1

  startListening(onNetworkAvailable: () => void) {
    this.callback = onNetworkAvailable
    this.netAvailableId = connection.on('netAvailable', () => { this.callback?.() })
    this.netUnavailableId = connection.on('netUnavailable', () => {})
  }

  stopListening() {
    if (this.netAvailableId >= 0) { connection.off('netAvailable', this.netAvailableId); this.netAvailableId = -1 }
    if (this.netUnavailableId >= 0) { connection.off('netUnavailable', this.netUnavailableId); this.netUnavailableId = -1 }
    this.callback = null
  }

  isNetworkAvailable(): boolean {
    try { return connection.getDefaultNetSync() !== null } catch { return false }
  }

  getNetworkType(): string {
    try {
      const netHandle = connection.getDefaultNetSync()
      if (!netHandle) return 'None'
      const netCap = connection.getNetCapabilitiesSync(netHandle)
      if (netCap.bearerTypes.includes(connection.NetBearType.BEARER_WIFI)) return 'WiFi'
      if (netCap.bearerTypes.includes(connection.NetBearType.BEARER_CELLULAR)) return 'Mobile'
      return 'Unknown'
    } catch { return 'None' }
  }
}
```

```typescript
// ========== MqttBackgroundService.ets ==========
import backgroundTaskManager from '@ohos.ability.backgroundTaskManager'
import { MqttClientManager } from '../mqtt/MqttClient'
import { ReconnectManager } from '../mqtt/ReconnectManager'
import { CommandHandler } from '../command/CommandHandler'
import { CommandMessage } from '../command/CommandMessage'
import { HeartbeatAdjuster } from '../battery/HeartbeatAdjuster'
import { NetworkMonitor } from '../battery/NetworkMonitor'

const TOPIC_HEARTBEAT = 'mdm/{deviceId}/heartbeat'
const TOPIC_RESULT = 'mdm/command/{deviceId}'

export class MqttBackgroundService {
  private mqttClient = new MqttClientManager()
  private reconnectManager = new ReconnectManager()
  private commandHandler = new CommandHandler()
  private heartbeatAdjuster = new HeartbeatAdjuster()
  private networkMonitor = new NetworkMonitor()
  private deviceId: string = ''
  private heartbeatTimerId: number = -1
  private reconnectTimerId: number = -1
  private backgroundTaskId: number = -1

  start(deviceId: string) {
    this.deviceId = deviceId
    this.requestBackgroundTask()
    this.networkMonitor.startListening(() => { if (this.networkMonitor.isNetworkAvailable()) this.scheduleReconnect(0) })
    this.mqttClient.addConnectionListener((connected: boolean) => {
      if (!connected) { this.reconnectManager.recordDisconnect(); this.scheduleReconnect(this.reconnectManager.getDelayMs()) }
      else this.reconnectManager.reset()
    })
    this.mqttClient.initialize(deviceId)
    this.mqttClient.setMessageHandler(this.onCommandMessage.bind(this))
    this.mqttClient.connect()
    this.startHeartbeat()
  }

  stop() {
    this.stopHeartbeat()
    this.networkMonitor.stopListening()
    this.reconnectManager.reset()
    this.mqttClient.disconnect()
    if (this.backgroundTaskId >= 0) backgroundTaskManager.cancelContinuousTask(this.backgroundTaskId)
  }

  private requestBackgroundTask() {
    try {
      const info = backgroundTaskManager.requestContinuousTaskInfo({ abilityName: 'MqttBackgroundService', persistent: true })
      this.backgroundTaskId = info.id
    } catch (e) { console.error('Background task request failed:', e) }
  }

  private scheduleReconnect(delayMs: number) {
    if (this.reconnectTimerId >= 0) clearTimeout(this.reconnectTimerId)
    this.reconnectTimerId = setTimeout(() => {
      if (!this.mqttClient.isConnectedState() && this.networkMonitor.isNetworkAvailable()) this.mqttClient.connect()
    }, delayMs) as unknown as number
  }

  private startHeartbeat() {
    const interval = this.heartbeatAdjuster.getIntervalMs()
    this.heartbeatTimerId = setInterval(() => { this.sendHeartbeat() }, interval) as unknown as number
    this.sendHeartbeat()
  }

  private stopHeartbeat() {
    if (this.heartbeatTimerId >= 0) { clearInterval(this.heartbeatTimerId); this.heartbeatTimerId = -1 }
  }

  private sendHeartbeat() {
    if (!this.mqttClient.isConnectedState()) return
    const topic = TOPIC_HEARTBEAT.replace('{deviceId}', this.deviceId)
    this.mqttClient.publish(topic, this.heartbeatAdjuster.buildHeartbeatMsg(this.deviceId))
  }

  private onCommandMessage(topic: string, payload: string) {
    const cmd = CommandMessage.fromJson(payload)
    if (!cmd) return
    console.info(`Processing command: ${cmd.action} (${cmd.commandId})`)
    this.reportResult(this.commandHandler.handle(cmd))
  }

  private reportResult(result: { toJson: () => string }) {
    const topic = TOPIC_RESULT.replace('{deviceId}', this.deviceId)
    this.mqttClient.publish(topic, result.toJson())
  }

  isConnected(): boolean { return this.mqttClient.isConnectedState() }
  getHeartbeatMode(): string { return this.heartbeatAdjuster.getHeartbeatMode() }
  getDeviceId(): string { return this.deviceId }
}
```
