import battery from '@ohos.battery'
import { hilog } from '@kit.PerformanceAnalysisKit'

const TAG = 'BatteryMonitor'
const DOMAIN = 0x0001

/**
 * 电池信息
 */
export interface BatteryInfo {
  batterySOC: number        // 电量百分比
  chargingStatus: number    // 充电状态
  isCharging: boolean       // 是否充电中
  isScreenOn: boolean        // 屏幕是否亮起
}

/**
 * 电池状态监听器
 */
export type BatteryChangeListener = (info: BatteryInfo) => void

/**
 * 电池监控器
 * 监听电池状态变化，提供动态心跳间隔计算
 */
export class BatteryMonitor {
  private listeners: BatteryChangeListener[] = []
  private lastInfo: BatteryInfo | null = null

  /**
   * 获取当前电池信息
   */
  getInfo(): BatteryInfo {
    try {
      const batteryInfo = battery.getBatteryInfo()
      const info: BatteryInfo = {
        batterySOC: batteryInfo.batterySOC,
        chargingStatus: batteryInfo.chargingStatus,
        isCharging: batteryInfo.isCharging,
        isScreenOn: this.isScreenOn()
      }
      this.lastInfo = info
      return info
    } catch (err) {
      hilog.error(DOMAIN, TAG, 'Failed to get battery info: %{public}s', JSON.stringify(err))
      // 返回默认值
      return {
        batterySOC: 100,
        chargingStatus: 0,
        isCharging: false,
        isScreenOn: true
      }
    }
  }

  /**
   * 检查屏幕是否亮起
   * 注意: 此API可能需要额外权限或在不同设备上行为不同
   */
  private isScreenOn(): boolean {
    try {
      // 尝试获取屏幕状态
      // 注意: @ohos.power的API可能因设备而异
      // import power from '@ohos.power'
      // return power.isScreenOn()
      return true // 默认返回true，如果API不可用
    } catch {
      return true
    }
  }

  /**
   * 监听电池状态变化
   * @param listener 监听回调
   */
  onChange(listener: BatteryChangeListener): void {
    this.listeners.push(listener)

    // 立即通知当前状态
    const info = this.getInfo()
    listener(info)
  }

  /**
   * 取消监听
   * @param listener 监听回调
   */
  off(listener: BatteryChangeListener): void {
    const index = this.listeners.indexOf(listener)
    if (index > -1) {
      this.listeners.splice(index, 1)
    }
  }

  /**
   * 通知所有监听器
   */
  private notifyListeners(): void {
    const info = this.getInfo()
    this.listeners.forEach(listener => {
      try {
        listener(info)
      } catch (err) {
        hilog.error(DOMAIN, TAG, 'Battery listener error: %{public}s', JSON.stringify(err))
      }
    })
  }

  /**
   * 是否处于低电量状态
   */
  isLowBattery(): boolean {
    const info = this.getInfo()
    return info.batterySOC < 20 && !info.isCharging
  }

  /**
   * 是否处于省电模式（基于电池状态）
   */
  isPowerSaving(): boolean {
    const info = this.getInfo()
    // 低电量或非充电状态且电量低于50%
    return info.batterySOC < 20 || (!info.isCharging && info.batterySOC < 50)
  }

  /**
   * 获取上一次获取的信息
   */
  getLastInfo(): BatteryInfo | null {
    return this.lastInfo
  }
}

import { BatteryMonitor, BatteryInfo } from './BatteryMonitor'
import { NetworkMonitor, NetworkInfo, NetworkType } from '../network/NetworkMonitor'
import { MqttConfig } from '../mqtt/MqttConfig'
import { hilog } from '@kit.PerformanceAnalysisKit'

const TAG = 'HeartbeatAdjuster'
const DOMAIN = 0x0001

/**
 * 心跳间隔模式
 */
export enum HeartbeatMode {
  HIGH_PERFORMANCE = 'high_performance',   // 高性能模式（充电中+电量充足）
  NORMAL = 'normal',                         // 正常模式
  POWER_SAVING = 'power_saving',             // 省电模式
  ULTRA_POWER_SAVING = 'ultra_power_saving'  // 极致省电模式
}

/**
 * 心跳消息结构
 */
interface HeartbeatMsg {
  deviceId: string
  batteryLevel: number
  isCharging: boolean
  isScreenOn: boolean
  networkType: NetworkType
  heartbeatMode: HeartbeatMode
  timestamp: number
}

/**
 * 心跳调整器
 * 根据电池状态、网络状态动态调整心跳间隔
 */
export class HeartbeatAdjuster {
  private batteryMonitor: BatteryMonitor
  private networkMonitor: NetworkMonitor

  constructor() {
    this.batteryMonitor = new BatteryMonitor()
    this.networkMonitor = new NetworkMonitor()
  }

  /**
   * 获取心跳间隔（毫秒）
   * 根据电量、充电状态、屏幕状态动态调整
   */
  getIntervalMs(): number {
    const batteryInfo = this.batteryMonitor.getInfo()
    const networkInfo = this.networkMonitor.getInfo()
    const mode = this.getMode()

    let interval: number

    switch (mode) {
      case HeartbeatMode.HIGH_PERFORMANCE:
        interval = MqttConfig.HEARTBEAT_HIGH_PERFORMANCE // 1分钟
        break
      case HeartbeatMode.NORMAL:
        interval = MqttConfig.HEARTBEAT_NORMAL // 2分钟
        break
      case HeartbeatMode.POWER_SAVING:
        interval = MqttConfig.HEARTBEAT_POWER_SAVING // 5分钟
        break
      case HeartbeatMode.ULTRA_POWER_SAVING:
        interval = MqttConfig.HEARTBEAT_ULTRA_POWER_SAVING // 10分钟
        break
      default:
        interval = MqttConfig.HEARTBEAT_NORMAL
    }

    hilog.info(DOMAIN, TAG, 'Heartbeat interval: %{public}dms (mode: %{public}s)', interval, mode)
    return interval
  }

  /**
   * 获取当前的心跳模式
   */
  getMode(): HeartbeatMode {
    const batteryInfo = this.batteryMonitor.getInfo()

    // 高性能模式：充电中且电量>50%
    if (batteryInfo.isCharging && batteryInfo.batterySOC > 50) {
      return HeartbeatMode.HIGH_PERFORMANCE
    }

    // 极致省电模式：息屏且未充电
    if (!batteryInfo.isScreenOn && !batteryInfo.isCharging) {
      return HeartbeatMode.ULTRA_POWER_SAVING
    }

    // 省电模式：电量<20%且未充电
    if (batteryInfo.batterySOC < 20 && !batteryInfo.isCharging) {
      return HeartbeatMode.POWER_SAVING
    }

    // 正常模式
    return HeartbeatMode.NORMAL
  }

  /**
   * 获取模式描述
   */
  getModeDescription(): string {
    switch (this.getMode()) {
      case HeartbeatMode.HIGH_PERFORMANCE:
        return '高性能模式 (充电中)'
      case HeartbeatMode.NORMAL:
        return '正常模式'
      case HeartbeatMode.POWER_SAVING:
        return '省电模式'
      case HeartbeatMode.ULTRA_POWER_SAVING:
        return '极致省电模式 (息屏)'
      default:
        return '未知模式'
    }
  }

  /**
   * 构建心跳消息
   * @param deviceId 设备ID
   */
  buildHeartbeatMsg(deviceId: string): string {
    const batteryInfo = this.batteryMonitor.getInfo()
    const networkInfo = this.networkMonitor.getInfo()

    const msg: HeartbeatMsg = {
      deviceId: deviceId,
      batteryLevel: batteryInfo.batterySOC,
      isCharging: batteryInfo.isCharging,
      isScreenOn: batteryInfo.isScreenOn,
      networkType: networkInfo.networkType,
      heartbeatMode: this.getMode(),
      timestamp: Date.now()
    }

    hilog.info(DOMAIN, TAG, 'Building heartbeat: %{public}s', JSON.stringify(msg))
    return JSON.stringify(msg)
  }

  /**
   * 获取电池监控器
   */
  getBatteryMonitor(): BatteryMonitor {
    return this.batteryMonitor
  }

  /**
   * 获取网络监控器
   */
  getNetworkMonitor(): NetworkMonitor {
    return this.networkMonitor
  }

  /**
   * 获取电池信息
   */
  getBatteryInfo(): BatteryInfo {
    return this.batteryMonitor.getInfo()
  }

  /**
   * 获取网络信息
   */
  getNetworkInfo(): NetworkInfo {
    return this.networkMonitor.getInfo()
  }
}

import { CommandMessage } from './CommandParser'
import { CommandResult } from './CommandResult'
import { MdmApiFactory } from '../mdm/MdmApiFactory'
import { hilog } from '@kit.PerformanceAnalysisKit'

const TAG = 'CommandHandler'
const DOMAIN = 0x0001

/**
 * 命令处理器
 * 负责解析MQTT消息并分发到对应的MDM模块执行
 */
export class CommandHandler {
  private deviceCtrl = MdmApiFactory.getDeviceController()
  private appMgr = MdmApiFactory.getAppManager()
  private restrictMgr = MdmApiFactory.getRestrictionManager()
  private infoQuerier = MdmApiFactory.getInfoQuerier()

  /**
   * 处理命令
   * @param cmd 命令消息
   * @returns 命令执行结果
   */
  async handle(cmd: CommandMessage): Promise<CommandResult> {
    hilog.info(DOMAIN, TAG, 'Handling command: %{public}s', cmd.toString())

    try {
      let result: CommandResult

      switch (cmd.action) {
        // 设备控制命令
        case 'lock':
          result = await this.handleLock(cmd)
          break
        case 'unlock':
          result = await this.handleUnlock(cmd)
          break
        case 'wipe':
          result = await this.handleWipe(cmd)
          break
        case 'reboot':
          result = await this.handleReboot(cmd)
          break
        case 'reset_password':
          result = await this.handleResetPassword(cmd)
          break

        // 应用管理命令
        case 'install_app':
          result = await this.handleInstallApp(cmd)
          break
        case 'uninstall_app':
          result = await this.handleUninstallApp(cmd)
          break
        case 'get_app_list':
          result = await this.handleGetAppList(cmd)
          break

        // 限制策略命令
        case 'disable_wifi':
          result = await this.handleDisableWifi(cmd, true)
          break
        case 'enable_wifi':
          result = await this.handleDisableWifi(cmd, false)
          break
        case 'disable_mobile_data':
          result = await this.handleDisableMobileData(cmd, true)
          break
        case 'enable_mobile_data':
          result = await this.handleDisableMobileData(cmd, false)
          break
        case 'set_gps':
          result = await this.handleSetGps(cmd)
          break
        case 'set_bluetooth':
          result = await this.handleSetBluetooth(cmd)
          break
        case 'disable_camera':
          result = await this.handleDisableCamera(cmd, true)
          break
        case 'enable_camera':
          result = await this.handleDisableCamera(cmd, false)
          break
        case 'disable_microphone':
          result = await this.handleDisableMicrophone(cmd, true)
          break
        case 'enable_microphone':
          result = await this.handleDisableMicrophone(cmd, false)
          break
        case 'disable_usb_debug':
          result = await this.handleDisableUsbDebug(cmd, true)
          break
        case 'enable_usb_debug':
          result = await this.handleDisableUsbDebug(cmd, false)
          break
        case 'disable_screenshot':
          result = await this.handleDisableScreenshot(cmd, true)
          break
        case 'enable_screenshot':
          result = await this.handleDisableScreenshot(cmd, false)
          break

        // 信息查询命令
        case 'get_device_info':
          result = await this.handleGetDeviceInfo(cmd)
          break

        // 未知命令
        default:
          hilog.warn(DOMAIN, TAG, 'Unknown command action: %{public}s', cmd.action)
          result = CommandResult.failure(cmd.commandId, cmd.action, `未知命令: ${cmd.action}`)
      }

      hilog.info(DOMAIN, TAG, 'Command handled: %{public}s', result.toString())
      return result
    } catch (err) {
      hilog.error(DOMAIN, TAG, 'Command handling error: %{public}s', JSON.stringify(err))
      return CommandResult.failure(cmd.commandId, cmd.action, `执行异常: ${err}`)
    }
  }

  // 设备控制处理

  private async handleLock(cmd: CommandMessage): Promise<CommandResult> {
    const mainText = cmd.getString('mainText', '设备已锁定')
    const subText = cmd.getString('subText', '请联系管理员')
    const res = await this.deviceCtrl.lock(mainText, subText)
    return new CommandResult(cmd.commandId, cmd.action, res.success, res.message)
  }

  private async handleUnlock(cmd: CommandMessage): Promise<CommandResult> {
    const res = await this.deviceCtrl.unlock()
    return new CommandResult(cmd.commandId, cmd.action, res.success, res.message)
  }

  private async handleWipe(cmd: CommandMessage): Promise<CommandResult> {
    const res = await this.deviceCtrl.wipe()
    return new CommandResult(cmd.commandId, cmd.action, res.success, res.message)
  }

  private async handleReboot(cmd: CommandMessage): Promise<CommandResult> {
    const res = await this.deviceCtrl.reboot()
    return new CommandResult(cmd.commandId, cmd.action, res.success, res.message)
  }

  private async handleResetPassword(cmd: CommandMessage): Promise<CommandResult> {
    const password = cmd.getString('password')
    if (!password) {
      return CommandResult.failure(cmd.commandId, cmd.action, '缺少password参数')
    }
    const res = await this.deviceCtrl.resetPassword(password)
    return new CommandResult(cmd.commandId, cmd.action, res.success, res.message)
  }

  // 应用管理处理

  private async handleInstallApp(cmd: CommandMessage): Promise<CommandResult> {
    const apkPath = cmd.getString('apkPath')
    if (!apkPath) {
      return CommandResult.failure(cmd.commandId, cmd.action, '缺少apkPath参数')
    }
    const res = await this.appMgr.install(apkPath)
    return new CommandResult(cmd.commandId, cmd.action, res.success, res.message)
  }

  private async handleUninstallApp(cmd: CommandMessage): Promise<CommandResult> {
    const bundleName = cmd.getString('bundleName')
    if (!bundleName) {
      return CommandResult.failure(cmd.commandId, cmd.action, '缺少bundleName参数')
    }
    const res = await this.appMgr.uninstall(bundleName)
    return new CommandResult(cmd.commandId, cmd.action, res.success, res.message)
  }

  private async handleGetAppList(cmd: CommandMessage): Promise<CommandResult> {
    const res = await this.appMgr.getAppList()
    return new CommandResult(cmd.commandId, cmd.action, res.success, res.message, { apps: res.apps })
  }

  // 限制策略处理

  private async handleDisableWifi(cmd: CommandMessage, disabled: boolean): Promise<CommandResult> {
    const res = await this.restrictMgr.setWifiDisabled(disabled)
    return new CommandResult(cmd.commandId, cmd.action, res.success, res.message)
  }

  private async handleDisableMobileData(cmd: CommandMessage, disabled: boolean): Promise<CommandResult> {
    const res = await this.restrictMgr.setMobileDataDisabled(disabled)
    return new CommandResult(cmd.commandId, cmd.action, res.success, res.message)
  }

  private async handleSetGps(cmd: CommandMessage): Promise<CommandResult> {
    const enabled = cmd.getBoolean('enabled', false)
    const res = await this.restrictMgr.setGpsEnabled(enabled)
    return new CommandResult(cmd.commandId, cmd.action, res.success, res.message)
  }

  private async handleSetBluetooth(cmd: CommandMessage): Promise<CommandResult> {
    const enabled = cmd.getBoolean('enabled', false)
    const res = await this.restrictMgr.setBluetoothDisabled(!enabled)
    return new CommandResult(cmd.commandId, cmd.action, res.success, res.message)
  }

  private async handleDisableCamera(cmd: CommandMessage, disabled: boolean): Promise<CommandResult> {
    const res = await this.restrictMgr.setCameraDisabled(disabled)
    return new CommandResult(cmd.commandId, cmd.action, res.success, res.message)
  }

  private async handleDisableMicrophone(cmd: CommandMessage, disabled: boolean): Promise<CommandResult> {
    const res = await this.restrictMgr.setMicrophoneDisabled(disabled)
    return new CommandResult(cmd.commandId, cmd.action, res.success, res.message)
  }

  private async handleDisableUsbDebug(cmd: CommandMessage, disabled: boolean): Promise<CommandResult> {
    const res = await this.restrictMgr.setUsbDebugDisabled(disabled)
    return new CommandResult(cmd.commandId, cmd.action, res.success, res.message)
  }

  private async handleDisableScreenshot(cmd: CommandMessage, disabled: boolean): Promise<CommandResult> {
    const res = await this.restrictMgr.setScreenshotDisabled(disabled)
    return new CommandResult(cmd.commandId, cmd.action, res.success, res.message)
  }

  // 信息查询处理

  private async handleGetDeviceInfo(cmd: CommandMessage): Promise<CommandResult> {
    const res = await this.infoQuerier.getDeviceInfo()
    return new CommandResult(cmd.commandId, cmd.action, res.success, res.message, res.data ? { info: res.data } : null)
  }
}

import { hilog } from '@kit.PerformanceAnalysisKit'

const TAG = 'CommandParser'
const DOMAIN = 0x0001

/**
 * 命令参数字典
 */
interface CommandParams {
  [key: string]: string | number | boolean | object | null
}

/**
 * 命令消息类
 * 解析和存储MQTT接收到的命令
 */
export class CommandMessage {
  /** 命令唯一标识 */
  commandId: string = ''
  /** 命令动作 */
  action: string = ''
  /** 命令参数 */
  params: CommandParams | null = null
  /** 时间戳 */
  timestamp: number = 0

  /**
   * 从JSON字符串解析命令
   * @param json JSON字符串
   * @returns 命令消息对象，解析失败返回null
   */
  static fromJson(json: string): CommandMessage | null {
    try {
      const obj = JSON.parse(json)

      if (!obj.commandId || !obj.action) {
        hilog.error(DOMAIN, TAG, 'Missing commandId or action in command')
        return null
      }

      const cmd = new CommandMessage()
      cmd.commandId = obj.commandId
      cmd.action = obj.action
      cmd.params = obj.params || null
      cmd.timestamp = obj.timestamp || Date.now()

      hilog.info(DOMAIN, TAG, 'Parsed command: %{public}s, action: %{public}s', cmd.commandId, cmd.action)
      return cmd
    } catch (err) {
      hilog.error(DOMAIN, TAG, 'Failed to parse command: %{public}s', JSON.stringify(err))
      return null
    }
  }

  /**
   * 获取字符串参数
   * @param key 参数键
   * @param def 默认值
   */
  getString(key: string, def: string = ''): string {
    if (!this.params || this.params[key] === undefined) {
      return def
    }
    return String(this.params[key])
  }

  /**
   * 获取布尔参数
   * @param key 参数键
   * @param def 默认值
   */
  getBoolean(key: string, def: boolean = false): boolean {
    if (!this.params || this.params[key] === undefined) {
      return def
    }
    return Boolean(this.params[key])
  }

  /**
   * 获取数字参数
   * @param key 参数键
   * @param def 默认值
   */
  getNumber(key: string, def: number = 0): number {
    if (!this.params || this.params[key] === undefined) {
      return def
    }
    const val = this.params[key]
    if (typeof val === 'number') {
      return val
    }
    const parsed = parseFloat(String(val))
    return isNaN(parsed) ? def : parsed
  }

  /**
   * 检查参数是否存在
   * @param key 参数键
   */
  hasParam(key: string): boolean {
    return this.params !== null && this.params[key] !== undefined
  }

  /**
   * 转换为字符串
   */
  toString(): string {
    return `CommandMessage(commandId=${this.commandId}, action=${this.action}, timestamp=${this.timestamp})`
  }
}

/**

interface ResultData {
  [key: string]: string | number | boolean | object | null
}

/**
 * 命令结果类
 * 封装命令执行结果，用于MQTT上报
 */
export class CommandResult {
  /** 命令唯一标识 */
  commandId: string = ''
  /** 命令动作 */
  action: string = ''
  /** 是否成功 */
  success: boolean = false
  /** 结果消息 */
  message: string = ''
  /** 附加数据 */
  data: ResultData | null = null
  /** 时间戳 */
  timestamp: number = 0

  constructor(
    commandId: string = '',
    action: string = '',
    success: boolean = false,
    message: string = '',
    data: ResultData | null = null
  ) {
    this.commandId = commandId
    this.action = action
    this.success = success
    this.message = message
    this.data = data
    this.timestamp = Date.now()
  }

  /**
   * 创建成功结果
   * @param cmdId 命令ID
   * @param action 命令动作
   * @param msg 结果消息
   * @param data 附加数据
   */
  static success(cmdId: string, action: string, msg: string = '执行成功', data?: ResultData): CommandResult {
    return new CommandResult(cmdId, action, true, msg, data || null)
  }

  /**
   * 创建失败结果
   * @param cmdId 命令ID
   * @param action 命令动作
   * @param msg 错误消息
   */
  static failure(cmdId: string, action: string, msg: string): CommandResult {
    return new CommandResult(cmdId, action, false, msg, null)
  }

  /**
   * 转换为JSON字符串
   */
  toJson(): string {
    return JSON.stringify({
      commandId: this.commandId,
      action: this.action,
      success: this.success,
      message: this.message,
      timestamp: this.timestamp,
      data: this.data
    })
  }

  /**
   * 从JSON字符串解析
   * @param json JSON字符串
   */
  static fromJson(json: string): CommandResult | null {
    try {
      const obj = JSON.parse(json)
      const result = new CommandResult()
      result.commandId = obj.commandId || ''
      result.action = obj.action || ''
      result.success = obj.success || false
      result.message = obj.message || ''
      result.data = obj.data || null
      result.timestamp = obj.timestamp || Date.now()
      return result
    } catch {
      return null
    }
  }

  /**
   * 转换为字符串（用于调试）
   */
  toString(): string {
    return `CommandResult(commandId=${this.commandId}, action=${this.action}, success=${this.success}, message=${this.message})`
  }
}

const DOMAIN = 0x0001

import mqtt from '@ohos/mqtt'
import { MqttConfig } from './MqttConfig'
import { ReconnectManager } from './ReconnectManager'
import { hilog } from '@kit.PerformanceAnalysisKit'

const TAG = 'MqttClientManager'
const DOMAIN = 0x0001

/**
 * MQTT连接状态
 */
export enum MqttConnectionState {
  DISCONNECTED = 'disconnected',
  CONNECTING = 'connecting',
  CONNECTED = 'connected',
  RECONNECTING = 'reconnecting',
  ERROR = 'error'
}

/**
 * MQTT客户端选项
 */
interface MqttClientOptions {
  clientId: string
  cleanSession: boolean
  keepAlive: number
}

/**
 * MQTT遗嘱消息选项
 */
interface WillOptions {
  topic: string
  payload: string
  qos: number
  retained: boolean
}

/**
 * MQTT连接选项
 */
interface ConnectOptions {
  username: string
  password: string
  connectTimeout: number
  reconnectPeriod: number
  will: WillOptions
}

/**
 * 命令数据接口
 */
interface CommandData {
  [key: string]: string | number | boolean | object | null
}

/**
 * MQTT客户端管理器
 * 负责MQTT连接、订阅、发布、重连等操作
 */
export class MqttClientManager {
  private client: mqtt.MqttClient | null = null
  private deviceId: string = ''
  private connectionState: MqttConnectionState = MqttConnectionState.DISCONNECTED
  private reconnectManager: ReconnectManager = new ReconnectManager()
  private reconnectTimer: number | null = null

  // 监听器
  private connectionListeners: Array<(state: MqttConnectionState) => void> = []
  private messageListeners: Array<(topic: string, payload: string) => void> = []

  /**
   * 初始化MQTT客户端
   * @param deviceId 设备唯一标识
   */
  initialize(deviceId: string): void {
    this.deviceId = deviceId
    hilog.info(DOMAIN, TAG, 'Initializing MQTT client for device: %{public}s', deviceId)

    // 创建MQTT客户端
    // 注意: 实际API可能因版本而异，这里使用常见写法
    try {
      const clientId = MqttConfig.generateClientId(deviceId)
      const options: MqttClientOptions = {
        clientId: clientId,
        cleanSession: MqttConfig.CLEAN_SESSION,
        keepAlive: MqttConfig.KEEP_ALIVE_INTERVAL
      }
      this.client = mqtt.createMqttClient(MqttConfig.SERVER_URI, options)

      // 设置事件监听
      this.setupEventHandlers()
      hilog.info(DOMAIN, TAG, 'MQTT client created successfully')
    } catch (err) {
      hilog.error(DOMAIN, TAG, 'Failed to create MQTT client: %{public}s', JSON.stringify(err))
      this.connectionState = MqttConnectionState.ERROR
      this.notifyConnectionListeners()
    }
  }

  /**
   * 设置事件处理器
   */
  private setupEventHandlers(): void {
    if (!this.client) return

    // 连接成功
    this.client.on('connect', () => {
      hilog.info(DOMAIN, TAG, 'MQTT connected')
      this.connectionState = MqttConnectionState.CONNECTED
      this.reconnectManager.reset()
      this.notifyConnectionListeners()
      // 订阅默认主题
      this.subscribeDefaultTopics()
    })

    // 连接关闭
    this.client.on('close', () => {
      hilog.warn(DOMAIN, TAG, 'MQTT connection closed')
      this.connectionState = MqttConnectionState.DISCONNECTED
      this.reconnectManager.recordDisconnect()
      this.notifyConnectionListeners()
    })

    // 连接错误
    this.client.on('error', (err: Error) => {
      hilog.error(DOMAIN, TAG, 'MQTT error: %{public}s', JSON.stringify(err))
      this.connectionState = MqttConnectionState.ERROR
      this.notifyConnectionListeners()
    })

    // 消息到达
    this.client.on('message', (topic: string, payload: string) => {
      hilog.info(DOMAIN, TAG, 'MQTT message received: %{public}s', topic)
      this.notifyMessageListeners(topic, payload)
    })

    // _disconnect事件(某些客户端实现)
    this.client.on('disconnect', () => {
      hilog.warn(DOMAIN, TAG, 'MQTT disconnected event')
      this.connectionState = MqttConnectionState.DISCONNECTED
      this.reconnectManager.recordDisconnect()
      this.notifyConnectionListeners()
    })
  }

  /**
   * 订阅默认主题
   */
  private subscribeDefaultTopics(): void {
    const commandTopic = MqttConfig.getCommandTopic(this.deviceId)
    const configTopic = MqttConfig.getConfigTopic(this.deviceId)

    this.subscribe(commandTopic, MqttConfig.SUBSCRIBE_QOS)
    this.subscribe(configTopic, MqttConfig.SUBSCRIBE_QOS)

    hilog.info(DOMAIN, TAG, 'Subscribed to default topics: %{public}s, %{public}s', commandTopic, configTopic)
  }

  /**
   * 连接到MQTT服务器
   */
  connect(): void {
    if (!this.client) {
      hilog.error(DOMAIN, TAG, 'MQTT client not initialized')
      return
    }

    if (this.connectionState === MqttConnectionState.CONNECTING) {
      hilog.warn(DOMAIN, TAG, 'MQTT already connecting')
      return
    }

    hilog.info(DOMAIN, TAG, 'Connecting to MQTT server: %{public}s', MqttConfig.SERVER_URI)
    this.connectionState = MqttConnectionState.CONNECTING
    this.notifyConnectionListeners()

    try {
      const willPayload: WillOptions = {
        topic: MqttConfig.WILL_TOPIC,
        payload: JSON.stringify({
          action: 'offline',
          deviceId: this.deviceId,
          timestamp: Date.now()
        }),
        qos: MqttConfig.WILL_QOS,
        retained: MqttConfig.WILL_RETAINED
      }
      const connectOptions: ConnectOptions = {
        username: MqttConfig.USERNAME,
        password: MqttConfig.PASSWORD,
        connectTimeout: MqttConfig.CONNECTION_TIMEOUT * 1000,
        reconnectPeriod: 0, // 我们使用自己的重连逻辑
        will: willPayload
      }
      this.client.connect(connectOptions)
    } catch (err) {
      hilog.error(DOMAIN, TAG, 'MQTT connect error: %{public}s', JSON.stringify(err))
      this.connectionState = MqttConnectionState.ERROR
      this.scheduleReconnect()
      this.notifyConnectionListeners()
    }
  }

  /**
   * 断开连接
   */
  disconnect(): void {
    hilog.info(DOMAIN, TAG, 'Disconnecting MQTT')
    this.cancelReconnect()
    this.reconnectManager.reset()

    if (this.client) {
      try {
        this.client.disconnect()
      } catch (err) {
        hilog.error(DOMAIN, TAG, 'MQTT disconnect error: %{public}s', JSON.stringify(err))
      }
    }

    this.connectionState = MqttConnectionState.DISCONNECTED
    this.notifyConnectionListeners()
  }

  /**
   * 订阅主题
   * @param topic 主题名
   * @param qos 服务质量等级
   */
  subscribe(topic: string, qos: number = MqttConfig.SUBSCRIBE_QOS): void {
    if (!this.client || !this.isConnected()) {
      hilog.error(DOMAIN, TAG, 'Cannot subscribe: client not connected')
      return
    }

    try {
      this.client.subscribe(topic, qos)
      hilog.info(DOMAIN, TAG, 'Subscribed to topic: %{public}s (QoS: %{public}d)', topic, qos)
    } catch (err) {
      hilog.error(DOMAIN, TAG, 'Subscribe error: %{public}s', JSON.stringify(err))
    }
  }

  /**
   * 取消订阅主题
   * @param topic 主题名
   */
  unsubscribe(topic: string): void {
    if (!this.client) return

    try {
      this.client.unsubscribe(topic)
      hilog.info(DOMAIN, TAG, 'Unsubscribed from topic: %{public}s', topic)
    } catch (err) {
      hilog.error(DOMAIN, TAG, 'Unsubscribe error: %{public}s', JSON.stringify(err))
    }
  }

  /**
   * 发布消息
   * @param topic 主题名
   * @param message 消息内容
   * @param qos 服务质量等级
   * @returns 是否发送成功
   */
  publish(topic: string, message: string, qos: number = MqttConfig.PUBLISH_QOS): boolean {
    if (!this.client || !this.isConnected()) {
      hilog.error(DOMAIN, TAG, 'Cannot publish: client not connected')
      return false
    }

    try {
      this.client.publish(topic, message, qos)
      hilog.info(DOMAIN, TAG, 'Published to topic: %{public}s', topic)
      return true
    } catch (err) {
      hilog.error(DOMAIN, TAG, 'Publish error: %{public}s', JSON.stringify(err))
      return false
    }
  }

  /**
   * 发布命令结果
   * @param commandId 命令ID
   * @param action 命令动作
   * @param success 是否成功
   * @param message 结果消息
   * @param data 附加数据
   */
  publishCommandResult(commandId: string, action: string, success: boolean, message: string, data?: CommandData): boolean {
    const resultTopic = MqttConfig.getResultTopic(this.deviceId)
    const payload = JSON.stringify({
      commandId,
      action,
      success,
      message,
      timestamp: Date.now(),
      data: data || {}
    })
    return this.publish(resultTopic, payload, MqttConfig.PUBLISH_QOS)
  }

  /**
   * 发布心跳
   * @param batteryLevel 电量
   * @param isCharging 是否充电
   * @param isScreenOn 屏幕是否亮起
   * @param networkType 网络类型
   */
  publishHeartbeat(batteryLevel: number, isCharging: boolean, isScreenOn: boolean, networkType: string): boolean {
    const heartbeatTopic = MqttConfig.getHeartbeatTopic(this.deviceId)
    const payload = JSON.stringify({
      deviceId: this.deviceId,
      batteryLevel,
      isCharging,
      isScreenOn,
      networkType,
      timestamp: Date.now()
    })
    return this.publish(heartbeatTopic, payload, 0) // 心跳使用QoS 0
  }

  /**
   * 发布告警
   * @param alertType 告警类型
   * @param alertMessage 告警消息
   * @param data 附加数据
   */
  publishAlert(alertType: string, alertMessage: string, data?: CommandData): boolean {
    const alertTopic = MqttConfig.getAlertTopic(this.deviceId)
    const payload = JSON.stringify({
      deviceId: this.deviceId,
      alertType,
      alertMessage,
      timestamp: Date.now(),
      data: data || {}
    })
    return this.publish(alertTopic, payload, MqttConfig.PUBLISH_QOS)
  }

  /**
   * 是否已连接
   */
  isConnected(): boolean {
    return this.connectionState === MqttConnectionState.CONNECTED
  }

  /**
   * 获取连接状态
   */
  getConnectionState(): MqttConnectionState {
    return this.connectionState
  }

  /**
   * 获取设备ID
   */
  getDeviceId(): string {
    return this.deviceId
  }

  /**
   * 添加连接状态监听器
   * @param listener 监听回调
   */
  addConnectionListener(listener: (state: MqttConnectionState) => void): void {
    this.connectionListeners.push(listener)
  }

  /**
   * 添加消息监听器
   * @param listener 监听回调
   */
  addMessageListener(listener: (topic: string, payload: string) => void): void {
    this.messageListeners.push(listener)
  }

  /**
   * 移除连接状态监听器
   * @param listener 监听回调
   */
  removeConnectionListener(listener: (state: MqttConnectionState) => void): void {
    const index = this.connectionListeners.indexOf(listener)
    if (index > -1) {
      this.connectionListeners.splice(index, 1)
    }
  }

  /**
   * 移除消息监听器
   * @param listener 监听回调
   */
  removeMessageListener(listener: (topic: string, payload: string) => void): void {
    const index = this.messageListeners.indexOf(listener)
    if (index > -1) {
      this.messageListeners.splice(index, 1)
    }
  }

  /**
   * 通知连接状态变化
   */
  private notifyConnectionListeners(): void {
    this.connectionListeners.forEach(listener => {
      try {
        listener(this.connectionState)
      } catch (err) {
        hilog.error(DOMAIN, TAG, 'Connection listener error: %{public}s', JSON.stringify(err))
      }
    })
  }

  /**
   * 通知消息到达
   */
  private notifyMessageListeners(topic: string, payload: string): void {
    this.messageListeners.forEach(listener => {
      try {
        listener(topic, payload)
      } catch (err) {
        hilog.error(DOMAIN, TAG, 'Message listener error: %{public}s', JSON.stringify(err))
      }
    })
  }

  /**
   * 计划重连
   */
  private scheduleReconnect(): void {
    this.cancelReconnect()

    if (!this.reconnectManager.shouldReconnect(0)) {
      hilog.warn(DOMAIN, TAG, 'Max reconnect count reached, stop reconnecting')
      return
    }

    const delay = this.reconnectManager.getDelayMs()
    hilog.info(DOMAIN, TAG, 'Scheduling reconnect in %{public}d ms', delay)

    this.connectionState = MqttConnectionState.RECONNECTING
    this.notifyConnectionListeners()

    this.reconnectTimer = setTimeout(() => {
      hilog.info(DOMAIN, TAG, 'Attempting reconnect...')
      this.connect()
    }, delay)
  }

  /**
   * 取消重连计划
   */
  private cancelReconnect(): void {
    if (this.reconnectTimer !== null) {
      clearTimeout(this.reconnectTimer)
      this.reconnectTimer = null
    }
  }

  /**
   * 释放资源
   */
  release(): void {
    hilog.info(DOMAIN, TAG, 'Releasing MQTT client')
    this.cancelReconnect()
    this.connectionListeners = []
    this.messageListeners = []

    if (this.client) {
      try {
        this.client.disconnect()
        // 某些客户端可能需要销毁
        // this.client.destroy?.()
      } catch (err) {
        hilog.error(DOMAIN, TAG, 'Release error: %{public}s', JSON.stringify(err))
      }
      this.client = null
    }

    this.connectionState = MqttConnectionState.DISCONNECTED
  }
}

import { hilog } from '@kit.PerformanceAnalysisKit'
import { MqttConfig } from './MqttConfig'

const TAG = 'ReconnectManager'
const DOMAIN = 0x0001

/**
 * MQTT重连管理器
 * 根据断开连接时长动态调整重连延迟
 */
export class ReconnectManager {
  private disconnectTime: number = 0  // 断开连接的时间戳
  private reconnectCount: number = 0  // 重连次数

  /**
   * 记录断开连接的时间
   */
  recordDisconnect(): void {
    this.disconnectTime = Date.now()
    this.reconnectCount++
    hilog.info(DOMAIN, TAG, 'Record disconnect at %{public}d, reconnect count: %{public}d',
      this.disconnectTime, this.reconnectCount)
  }

  /**
   * 获取重连延迟时间(毫秒)
   * 根据断开连接时长动态调整:
   * - < 5分钟: 5秒
   * - < 30分钟: 30秒
   * - < 60分钟: 1分钟
   * - < 360分钟: 3分钟
   * - >= 360分钟: 5分钟
   */
  getDelayMs(): number {
    if (this.disconnectTime === 0) {
      // 首次连接，使用快速重连
      return MqttConfig.RECONNECT_FAST
    }

    const minutes = (Date.now() - this.disconnectTime) / 60000

    let delay: number
    if (minutes < 5) {
      delay = MqttConfig.RECONNECT_FAST
    } else if (minutes < 30) {
      delay = MqttConfig.RECONNECT_NORMAL
    } else if (minutes < 60) {
      delay = MqttConfig.RECONNECT_SLOW
    } else if (minutes < 360) {
      delay = MqttConfig.RECONNECT_VERY_SLOW
    } else {
      delay = MqttConfig.RECONNECT_ULTRA_SLOW
    }

    hilog.info(DOMAIN, TAG, 'Get reconnect delay: %{public}dms (disconnected for %{public}f minutes)', delay, minutes)
    return delay
  }

  /**
   * 重置重连管理器
   * 连接成功后调用
   */
  reset(): void {
    const wasCount = this.reconnectCount
    this.disconnectTime = 0
    this.reconnectCount = 0
    hilog.info(DOMAIN, TAG, 'ReconnectManager reset, previous reconnect count: %{public}d', wasCount)
  }

  /**
   * 获取当前重连次数
   */
  getReconnectCount(): number {
    return this.reconnectCount
  }

  /**
   * 获取断开连接时长(毫秒)
   */
  getDisconnectDuration(): number {
    if (this.disconnectTime === 0) {
      return 0
    }
    return Date.now() - this.disconnectTime
  }

  /**
   * 是否应该继续重连
   * @param maxReconnectCount 最大重连次数，0表示无限制
   */
  shouldReconnect(maxReconnectCount: number = 0): boolean {
    if (maxReconnectCount === 0) {
      return true
    }
    return this.reconnectCount < maxReconnectCount
  }
}

export enum NetworkType {

  CELLULAR = 'Cellular',
  ETHERNET = 'Ethernet',

  UNKNOWN = 'Unknown'

/**
 * 网络状态信息
 */
export interface NetworkInfo {
  isAvailable: boolean       // 网络是否可用
  networkType: NetworkType   // 网络类型
  isMetered: boolean         // 是否按流量计费
}

/**
 * 网络状态监听器
 */
export type NetworkChangeListener = (info: NetworkInfo) => void

/**
 * 网络监控器
 * 监听网络状态变化
 */
export class NetworkMonitor {
  private listeners: NetworkChangeListener[] = []
  private lastInfo: NetworkInfo | null = null

  /**
   * 获取当前网络信息
   */
  getInfo(): NetworkInfo {
    try {
      const netHandle = connection.getDefaultNetSync()
      const isAvailable = netHandle !== null

      if (!isAvailable) {
        return {
          isAvailable: false,
          networkType: NetworkType.NONE,
          isMetered: false
        }
      }

      // 获取网络类型
      const netCapability = connection.getNetCapabilitiesSync(netHandle)
      let networkType = NetworkType.UNKNOWN

      if (netCapability.netCapabilities.includes(connection.NetCap.NET_CAPABILITY_WIFI)) {
        networkType = NetworkType.WIFI
      } else if (netCapability.netCapabilities.includes(connection.NetCap.NET_CAPABILITY_MOBILE)) {
        networkType = NetworkType.CELLULAR
      } else if (netCapability.netCapabilities.includes(connection.NetCap.NET_CAPABILITY_ETHERNET)) {
        networkType = NetworkType.ETHERNET
      }

      const info: NetworkInfo = {
        isAvailable: true,
        networkType: networkType,
        isMetered: netCapability.isMetered || false
      }

      this.lastInfo = info
      return info
    } catch (err) {
      hilog.error(DOMAIN, TAG, 'Failed to get network info: %{public}s', JSON.stringify(err))
      return {
        isAvailable: false,
        networkType: NetworkType.NONE,
        isMetered: false
      }
    }
  }

  /**
   * 检查网络是否可用
   */
  isNetworkAvailable(): boolean {
    return this.getInfo().isAvailable
  }

  /**
   * 监听网络状态变化
   * @param listener 监听回调
   */
  onChange(listener: NetworkChangeListener): void {
    this.listeners.push(listener)

    // 立即通知当前状态
    const info = this.getInfo()
    listener(info)

    // 注册网络状态变化回调
    try {
      connection.on('netAvailable', () => {
        hilog.info(DOMAIN, TAG, 'Network available changed')
        this.notifyListeners()
      })

      connection.on('netUnavailable', () => {
        hilog.info(DOMAIN, TAG, 'Network unavailable')
        this.notifyListeners()
      })
    } catch (err) {
      hilog.error(DOMAIN, TAG, 'Failed to register network listener: %{public}s', JSON.stringify(err))
    }
  }

  /**
   * 取消所有监听
   */
  off(): void {
    try {
      connection.off('netAvailable')
      connection.off('netUnavailable')
    } catch (err) {
      hilog.error(DOMAIN, TAG, 'Failed to unregister network listener: %{public}s', JSON.stringify(err))
    }
    this.listeners = []
  }

  /**
   * 通知所有监听器
   */
  private notifyListeners(): void {
    const info = this.getInfo()
    this.listeners.forEach(listener => {
      try {
        listener(info)
      } catch (err) {
        hilog.error(DOMAIN, TAG, 'Network listener error: %{public}s', JSON.stringify(err))
      }
    })
  }

  /**
   * 获取上一次获取的信息
   */
  getLastInfo(): NetworkInfo | null {
    return this.lastInfo
  }
}

import backgroundTaskManager from '@ohos.ability.backgroundTaskManager'
import { BusinessError } from '@ohos.base'
import { hilog } from '@kit.PerformanceAnalysisKit'
import { MqttClientManager, MqttConnectionState } from '../mqtt/MqttClient'
import { CommandHandler } from '../command/CommandHandler'
import { CommandMessage } from '../command/CommandParser'
import { HeartbeatAdjuster } from '../battery/HeartbeatAdjuster'
import { ConfigStore } from '../storage/ConfigStore'

const TAG = 'MqttBackgroundService'
const DOMAIN = 0x0001
const BACKGROUND_TASK_ID = 10001

/**
 * MQTT后台服务
 * 负责维护MQTT长连接、处理命令、心跳等后台任务
 */
export class MqttBackgroundService {
  private mqttClient: MqttClientManager | null = null
  private commandHandler: CommandHandler
  private heartbeatAdjuster: HeartbeatAdjuster
  private heartbeatTimer: number | null = null
  private isRunning: boolean = false
  private backgroundTaskId: number = 0

  constructor() {
    this.commandHandler = new CommandHandler()
    this.heartbeatAdjuster = new HeartbeatAdjuster()
  }

  /**
   * 启动后台服务
   * @param context 应用上下文
   */
  async start(context: Context): Promise<void> {
    if (this.isRunning) {
      hilog.warn(DOMAIN, TAG, 'Service already running')
      return
    }

    hilog.info(DOMAIN, TAG, 'Starting MQTT background service')

    try {
      // 初始化配置存储
      await ConfigStore.init(context)

      // 初始化MQTT客户端
      const deviceId = await ConfigStore.getDeviceId()
      this.mqttClient = new MqttClientManager()
      this.mqttClient.initialize(deviceId)

      // 设置连接状态监听
      this.mqttClient.addConnectionListener((state) => {
        this.onConnectionStateChanged(state)
      })

      // 设置消息监听
      this.mqttClient.addMessageListener((topic, payload) => {
        this.onMessageReceived(topic, payload)
      })

      // 申请后台任务
      this.requestBackgroundTask()

      // 启动心跳
      this.startHeartbeat()

      // 连接MQTT服务器
      this.mqttClient.connect()

      this.isRunning = true
      hilog.info(DOMAIN, TAG, 'MQTT background service started')
    } catch (err) {
      hilog.error(DOMAIN, TAG, 'Failed to start service: %{public}s', JSON.stringify(err))
      throw
    }
  }

  /**
   * 停止后台服务
   */
  stop(): void {
    if (!this.isRunning) {
      return
    }

    hilog.info(DOMAIN, TAG, 'Stopping MQTT background service')

    // 停止心跳
    this.stopHeartbeat()

    // 断开MQTT连接
    if (this.mqttClient) {
      this.mqttClient.disconnect()
      this.mqttClient.release()
      this.mqttClient = null
    }

    // 取消后台任务
    this.cancelBackgroundTask()

    this.isRunning = false
    hilog.info(DOMAIN, TAG, 'MQTT background service stopped')
  }

  /**
   * 申请后台任务
   */
  private requestBackgroundTask(): void {
    try {
      this.backgroundTaskId = backgroundTaskManager.requestBackgroundRunningprivilege(
        (err: BusinessError) => {
          if (err) {
            hilog.error(DOMAIN, TAG, 'Request background task failed: %{public}s', err.message)
          } else {
            hilog.info(DOMAIN, TAG, 'Background task granted')
          }
        }
      )
      hilog.info(DOMAIN, TAG, 'Background task ID: %{public}d', this.backgroundTaskId)
    } catch (err) {
      hilog.error(DOMAIN, TAG, 'Request background task error: %{public}s', JSON.stringify(err))
    }
  }

  /**
   * 取消后台任务
   */
  private cancelBackgroundTask(): void {
    if (this.backgroundTaskId !== 0) {
      try {
        backgroundTaskManager.cancelBackgroundRunningprivilege(this.backgroundTaskId)
        hilog.info(DOMAIN, TAG, 'Background task cancelled')
      } catch (err) {
        hilog.error(DOMAIN, TAG, 'Cancel background task error: %{public}s', JSON.stringify(err))
      }
    }
  }

  /**
   * 连接状态变化处理
   */
  private onConnectionStateChanged(state: MqttConnectionState): void {
    hilog.info(DOMAIN, TAG, 'Connection state changed: %{public}s', state)

    switch (state) {
      case MqttConnectionState.CONNECTED:
        hilog.info(DOMAIN, TAG, 'MQTT connected successfully')
        // 保存最后心跳时间
        ConfigStore.saveLastHeartbeatTime(Date.now())
        break
      case MqttConnectionState.DISCONNECTED:
        hilog.warn(DOMAIN, TAG, 'MQTT disconnected')
        break
      case MqttConnectionState.RECONNECTING:
        hilog.info(DOMAIN, TAG, 'MQTT reconnecting...')
        break
      case MqttConnectionState.ERROR:
        hilog.error(DOMAIN, TAG, 'MQTT error')
        break
    }
  }

  /**
   * 收到消息处理
   */
  private async onMessageReceived(topic: string, payload: string): Promise<void> {
    hilog.info(DOMAIN, TAG, 'Message received on topic: %{public}s', topic)

    try {
      // 解析命令
      const cmd = CommandMessage.fromJson(payload)
      if (!cmd) {
        hilog.error(DOMAIN, TAG, 'Failed to parse command message')
        return
      }

      // 处理命令
      const result = await this.commandHandler.handle(cmd)

      // 上报命令结果
      if (this.mqttClient) {
        this.mqttClient.publishCommandResult(
          result.commandId,
          result.action,
          result.success,
          result.message,
          result.data
        )
      }
    } catch (err) {
      hilog.error(DOMAIN, TAG, 'Error handling message: %{public}s', JSON.stringify(err))
    }
  }

  /**
   * 启动心跳
   */
  private startHeartbeat(): void {
    if (this.heartbeatTimer !== null) {
      return
    }

    hilog.info(DOMAIN, TAG, 'Starting heartbeat')

    const sendHeartbeat = () => {
      if (this.mqttClient && this.mqttClient.isConnected()) {
        const msg = this.heartbeatAdjuster.buildHeartbeatMsg(this.mqttClient.getDeviceId())
        this.mqttClient.publishHeartbeat(
          this.heartbeatAdjuster.getBatteryInfo().batterySOC,
          this.heartbeatAdjuster.getBatteryInfo().isCharging,
          this.heartbeatAdjuster.getBatteryInfo().isScreenOn,
          this.heartbeatAdjuster.getNetworkInfo().networkType
        )
        hilog.info(DOMAIN, TAG, 'Heartbeat sent')
      }

      // 计划下一次心跳
      const interval = this.heartbeatAdjuster.getIntervalMs()
      this.heartbeatTimer = setTimeout(sendHeartbeat, interval)
    }

    // 立即发送第一次心跳
    sendHeartbeat()
  }

  /**
   * 停止心跳
   */
  private stopHeartbeat(): void {
    if (this.heartbeatTimer !== null) {
      clearTimeout(this.heartbeatTimer)
      this.heartbeatTimer = null
      hilog.info(DOMAIN, TAG, 'Heartbeat stopped')
    }
  }

  /**
   * 获取服务运行状态
   */
  isServiceRunning(): boolean {
    return this.isRunning
  }

  /**
   * 获取MQTT连接状态
   */
  getConnectionState(): MqttConnectionState | null {
    return this.mqttClient?.getConnectionState() || null
  }

  /**
   * 获取心跳调整器
   */
  getHeartbeatAdjuster(): HeartbeatAdjuster {
    return this.heartbeatAdjuster
  }
}

  static async init(context: Context): Promise<void> {
    try {
      contextInstance = context
      prefInstance = await preferences.getPreferences(context, PREFERENCES_NAME)

    }

  /**
   * 设置配置项
   * @param key 键
   * @param value 值
   */
  static async set(key: string, value: string): Promise<void> {
    if (!prefInstance) {
      hilog.error(DOMAIN, TAG, 'ConfigStore not initialized')
      return
    }
    try {
      await prefInstance.put(key, value)
      await prefInstance.flush()
      hilog.info(DOMAIN, TAG, 'Config set: %{public}s = %{public}s', key, value)
    } catch (err) {
      hilog.error(DOMAIN, TAG, 'Failed to set config: %{public}s', JSON.stringify(err))
    }

   * 获取配置项

    if (!prefInstance) {
      hilog.error(DOMAIN, TAG, 'ConfigStore not initialized')
      return def
    }
    try {
      const value = await prefInstance.get(key, def)
      if (typeof value === 'string') {
        return value
      }
      return def
    } catch (err) {
      hilog.error(DOMAIN, TAG, 'Failed to get config: %{public}s', JSON.stringify(err))
